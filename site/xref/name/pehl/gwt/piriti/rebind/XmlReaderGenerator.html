11  /**
12   * GWT Generator for XmlReaders. The generator delegates to
13   * {@link XmlReaderCreator} which is responsible for generating the
14   * implementation.
15   * 
16   * @author $LastChangedBy: harald.pehl $
17   * @version $LastChangedRevision: 9 $
18   */
19  public class XmlReaderGenerator extends Generator
20  {
21      @Override
22      public String generate(TreeLogger logger, GeneratorContext context, String typeName)
23              throws UnableToCompleteException
24      {
25          JClassType interfaceType;
26          TypeOracle oracle = context.getTypeOracle();
27          try
28          {
29              interfaceType = oracle.getType(typeName);
30          }
31          catch (NotFoundException e)
32          {
33              throw new RuntimeException(e);
34          }
35  
36          String implName = interfaceType.getName().replace('.', '_') + "Impl";
37          String packageName = interfaceType.getPackage().getName();
38          XmlReaderCreator xmlReaderwriter = new XmlReaderCreator(context, interfaceType, implName, logger);
39          xmlReaderwriter.create();
40          return packageName + "." + implName;
41      }
42  }
