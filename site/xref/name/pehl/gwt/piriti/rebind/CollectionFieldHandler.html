44       */
45      @Override
46      public boolean isValid(IndentedWriter writer, FieldContext fieldContext) throws UnableToCompleteException
47      {
48          if (!fieldContext.isCollection())
49          {
50              skipField(writer, fieldContext, "Type is no collection");
51              return false;
52          }
53          JParameterizedType parameterizedType = fieldContext.getFieldType().isParameterized();
54          if (parameterizedType != null)
55          {
56              JClassType[] typeArgs = parameterizedType.getTypeArgs();
57              for (JClassType typeArg : typeArgs)
58              {
59                  if (fieldContext.getModelType().equals(typeArg))
60                  {
61                      skipField(writer, fieldContext, "Type argument of the collection equals the model type");
62                      return false;
63                  }
64                  if (typeArg.isArray() != null || TypeUtils.isCollection(typeArg) || TypeUtils.isMap(typeArg))
65                  {
66                      skipField(writer, fieldContext, "Nested arrays / collections / maps are not supported");
67                      return false;
68                  }
69              }
70          }
71          else
72          {
73              // collections and maps without type arguments are not
74              // supported!
75              skipField(writer, fieldContext, "Collection has no type argument");
76              return false;
77          }
78          return true;
79      }
