56      /**
57       * Reads and maps the models from the response and calls either
58       * {@link #onSuccess(List, Request, Response)} or
59       * {@link #onError(IOException, Request, Response)}.
60       * 
61       * @param request
62       * @param response
63       * @see org.restlet.client.Uniform#handle(org.restlet.client.Request,
64       *      org.restlet.client.Response)
65       */
66      @Override
67      public void handle(Request request, Response response)
68      {
69          PiritiJsonRepresentation<T> representation = new PiritiJsonRepresentation<T>(jsonReader, response.getEntity());
70          try
71          {
72              List<T> models = representation.getModels();
73              onSuccess(models, request, response);
74          }
75          catch (IOException e)
76          {
77              onError(e, request, response);
78          }
79      }
