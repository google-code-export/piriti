25      /**
26       * Returns a handler based on the given {@link PropertyContext}. The
27       * following rules are implemented:
28       * <ol>
29       * <li>IDs: {@link #newIdHandler()}
30       * <li>IDREFs: {@link #newIdRefHandler()}
31       * <li>boolean, Boolean: {@link #newBooleanHandler()}
32       * <li>byte, short, int, long, float, double, Number:
33       * {@link #newNumericHandler()}
34       * <li>char, Character: {@link #newCharacterHandler()}
35       * <li>String: {@link #newStringHandler()}
36       * <li>Arrays: {@link #newArrayHandler()}
37       * <li>Collections: {@link #newCollectionHandler()}
38       * <li>any other type: {@link #newDefaultHandler()}
39       * </ol>
40       * 
41       * @param propertyContext
42       * @return
43       * @see name.pehl.piriti.rebind.propertyhandler.PropertyHandlerLookup#lookup(name.pehl.piriti.rebind.PropertyContext)
44       */
45      @Override
46      public PropertyHandler lookup(PropertyContext propertyContext)
47      {
48          PropertyHandler handler = null;
49          JType type = propertyContext.getType();
50          JPrimitiveType primitiveType = propertyContext.getPrimitiveType();
51  
52          if (propertyContext.getReferenceType() == ID)
53          {
54              handler = newIdHandler();
55          }
56          else if (propertyContext.getReferenceType() == IDREF)
57          {
58              handler = newIdRefHandler();
59          }
60          else if (TypeUtils.isBoolean(type) || TypeUtils.isBoolean(primitiveType))
61          {
62              handler = newBooleanHandler();
63          }
64          else if (TypeUtils.isNumeric(type) || TypeUtils.isNumeric(primitiveType))
65          {
66              handler = newNumericHandler();
67          }
68          else if (TypeUtils.isCharacter(type) || TypeUtils.isCharacter(primitiveType))
69          {
70              handler = newCharacterHandler();
71          }
72          else if (TypeUtils.isString(type))
73          {
74              handler = newStringHandler();
75          }
76          else if (type.isEnum() != null)
77          {
78              handler = newEnumHandler();
79          }
80          else if (type.isArray() != null)
81          {
82              handler = newArrayHandler();
83          }
84          else if (TypeUtils.isCollection(type))
85          {
86              handler = newCollectionHandler();
87          }
88          else
89          {
90              handler = newDefaultHandler();
91          }
92          return handler;
93      }
