83      /**
84       * To ensure all necessary reader and writer are registered, this little
85       * helper method genereates a new instance of the specified type (only if
86       * the type provides a noarg constructor).
87       * 
88       * <pre>
89       * new &lt;specified type&gt;();
90       * </pre>
91       * 
92       * @param writer
93       * @param type
94       */
95      public static void readerWriterInitialization(IndentedWriter writer, JClassType type)
96      {
97          List<JClassType> concreteTypes = new ArrayList<JClassType>();
98          collectConcreteTypes(concreteTypes, type);
99          for (JClassType concreteType : concreteTypes)
100         {
101             writer.write(
102                     "new %1$s(); // if there are any reader / writer definitions in %1$s, this ensures they are registered",
103                     concreteType.getParameterizedQualifiedSourceName());
104         }
105     }

108     private static void collectConcreteTypes(List<JClassType> concreteTypes, JClassType type)
109     {
110         if (type != null)
111         {
112             if (type.isAbstract() || type.isInterface() != null)
113             {
114                 JClassType[] subtypes = type.getSubtypes();
115                 if (subtypes != null && subtypes.length != 0)
116                 {
117                     for (JClassType subtype : subtypes)
118                     {
119                         collectConcreteTypes(concreteTypes, subtype);
120                     }
121                 }
122             }
123             else
124             {
125                 if (!(TypeUtils.isJavaType(type) || TypeUtils.isGwtType(type)) && TypeUtils.isDefaultInstantiable(type))
126                 {
127                     concreteTypes.add(type);
128                 }
129             }
130         }
