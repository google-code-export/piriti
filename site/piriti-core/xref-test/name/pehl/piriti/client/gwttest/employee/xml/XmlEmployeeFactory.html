52      private static Element createEmployee(Document document, String id, String name, String bossRef, String[] teamRefs,
53              Element departmentElement, String departmentRef)
54      {
55          Element employee = document.createElement("employee");
56          employee.setAttribute("id", id);
57          createElementAndAppend(document, employee, "name", name);
58          if (bossRef != null)
59          {
60              Element boss = document.createElement("boss");
61              boss.setAttribute("ref", bossRef);
62              employee.appendChild(boss);
63          }
64          if (teamRefs != null)
65          {
66              Element team = document.createElement("team");
67              for (String teamRef : teamRefs)
68              {
69                  Element member = document.createElement("member");
70                  member.setAttribute("ref", teamRef);
71                  team.appendChild(member);
72              }
73              employee.appendChild(team);
74          }
75          if (departmentElement != null)
76          {
77              employee.appendChild(departmentElement);
78          }
79          else if (departmentRef != null)
80          {
81              Element departmentRefElement = document.createElement("department");
82              departmentRefElement.setAttribute("ref", departmentRef);
83              employee.appendChild(departmentRefElement);
84          }
85          return employee;
86      }

89      private static Element createDepartment(Document document, String id, String name, String employees)
