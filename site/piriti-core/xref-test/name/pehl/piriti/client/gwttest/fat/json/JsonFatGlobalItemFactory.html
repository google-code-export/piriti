67      private static String createFatGlobalItem(boolean withNestedFatGlobalItems)

69          StringBuilder json = new StringBuilder();

72          // Primitives and simple objects

89          appendKeyValueArray(json, "arrayOfIntegerObjects", false, true, "0", "1", "2");
90          appendKeyValueArray(json, "arrayOfStrings", true, true, "0", "1", "2");
91          if (withNestedFatGlobalItems)

93              appendFatGlobalItmes(json, "arrayOfFatGlobalItems", SIZE, true);

99          appendKeyValueArray(json, "collectionOfStrings", true, true, "0", "1", "2");
100         if (withNestedFatGlobalItems)

102             appendFatGlobalItmes(json, "collectionOfFatGlobalItems", SIZE, true);
103         }
104         appendSkinnyNestedItems(json, "collectionOfSkinnyNestedItems", SIZE, true);
105 
106         // Lists

108         appendKeyValueArray(json, "listOfStrings", true, true, "0", "1", "2");
109         if (withNestedFatGlobalItems)
110         {
111             appendFatGlobalItmes(json, "listOfFatGlobalItems", SIZE, true);
112         }
113         appendSkinnyNestedItems(json, "listOfSkinnyNestedItems", SIZE, true);
114 
115         // Sets

117         appendKeyValueArray(json, "setOfStrings", true, true, "0", "1", "2");
118         if (withNestedFatGlobalItems)
