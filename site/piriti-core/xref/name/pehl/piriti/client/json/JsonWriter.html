7   /**
8    * Interface for serializing an instance of T or a list of Ts to JSON data. The
9    * implementation for this interface is generated using deferred binding. All
10   * fields of T which are annotated with {@link JsonField} are handled by the
11   * generated JsonWriter implementation.
12   * <p>
13   * Please note:
14   * <ul>
15   * <li>Fields in T must not be private
16   * <li>Null values in T are generated into the resulting JSON data
17   * </ul>
18   * <p>
19   * The setup of the XmlWriter is inspired by the UiBinder and is typically
20   * specified as an inner class:
21   * 
22   * <pre>
23   * pubilc RunnableMessageContext
24   * {
25   *     interface Writer extends JsonWriter&lt;RunnableMessageContext&gt; {}
26   *     public static final Writer JSON_WRITER = GWT.create(Writer.class);
27   *     
28   *     // The fields of this POJO annotated with JsonField.
29   * }
30   * </pre>
31   * 
32   * JSON data can then be generated by calling
33   * 
34   * <pre>
35   * RunnableMessageContext rmc = new RunnableMessageContext();
36   * ...
37   * String json = RunnableMessageContext.JSON_WRITER.toJson(rmc);
38   * </pre>
39   * 
40   * @param <T>
41   *            The type
42   * @author $LastChangedBy: harald.pehl $
43   * @version $LastChangedRevision: 46 $
44   */
45  public interface JsonWriter<T>
46  {
47      /**
48       * Generates JSON data for the list of Ts according to the annotated fields
49       * in T. The resulting JSON data contains the instances ot T inside an array
50       * with the specified name:
51       * 
52       * <pre>
53       * List&lt;RunnableMessageContext&gt; rmcs = ...;
54       * String json = RunnableMessageContext.JSON_WRITER.toJson(rmcs, "data");
55       * // json will be somethig like {"data":[...]}
56       * </pre>
57       * 
58       * @param models
59       *            the list of Ts to serialize to JSON.
60       * @param arrayKey
61       *            the array key
62       * @return JSON data for the list of Ts or <code>null</code> if
63       *         {@code models} and/or {@code arrayKey} is <code>null</code>.
64       */
65      String toJson(@Nullable List<T> models, String arrayKey);
66  
67  
68      /**
69       * Generates JSON data for T according to the annotated fields in T.
70       * 
71       * @param model
72       *            the instance to serialize to JSON.
73       * @return JSON data for T or <code>null</code> if {@code model} is
74       *         <code>null</code>.
75       */
76      String toJson(@Nullable T model);
