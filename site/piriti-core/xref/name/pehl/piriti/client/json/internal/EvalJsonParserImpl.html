12   * Javascript eval().
13   * <p>
14   * <b>Please note:</b><br/>

16   * untrusted string to the {@link #parse(String)} method.
17   * 
18   * @author $Author: harald.pehl $
19   * @version $Date: 2010-07-30 23:01:45 +0200 (Fr, 30. Jul 2010) $ $Revision: 531
20   *          $
21   */
22  public class EvalJsonParserImpl implements JsonParser
23  {
24      @Override
25      public JSONObject parse(String text) throws JSONException
26      {
27          try
28          {
29              JSONValue value = JSONParser.parse(text);
30              return value != null ? value.isObject() : null;
31          }
32          catch (Throwable t)
33          {
34              throw new JSONException(t);
35          }
36      }
