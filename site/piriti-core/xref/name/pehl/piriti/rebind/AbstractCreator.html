261     private <T extends Annotation> void collectFields(JClassType type, List<JField> fields, Class<T> annotationClass)
262     {
263         // Superclass first please!
264         if (type == null)
265         {
266             return;
267         }
268         collectFields(type.getSuperclass(), fields, annotationClass);
269 
270         JField[] allFields = type.getFields();
271         if (allFields != null)
272         {
273             for (JField field : allFields)
274             {
275                 if (field.isAnnotationPresent(annotationClass))
276                 {
277                     fields.add(field);
278                 }
279             }
280         }
281     }
