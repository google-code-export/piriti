91      public PropertyContext(TypeOracle typeOracle, PropertyHandlerRegistry handlerRegistry, JClassType readerOrWriter,
92              JClassType clazz, JType type, String name, String path, String format, boolean stripWsnl,
93              Class<? extends Converter<?>> converter, ReferenceType referenceType,
94              Class<? extends PropertyGetter<?, ?>> getter, Class<? extends PropertySetter<?, ?>> setter,
95              VariableNames variableNames) throws UnableToCompleteException
96      {
97          // Types
98          this.typeOracle = typeOracle;
99          this.handlerRegistry = handlerRegistry;
100         this.readerOrWriter = readerOrWriter;
101         this.clazz = clazz;
102         JPrimitiveType primitiveType = type.isPrimitive();
103         if (primitiveType != null) // isPrimitive() is not yet available!
104         {
105             try
106             {
107                 // Use the boxed type for primitives
108                 this.type = typeOracle.getType(primitiveType.getQualifiedBoxedSourceName());
109                 this.primitiveType = primitiveType;
110             }
111             catch (NotFoundException e)
112             {
113                 throw new UnableToCompleteException();
114             }
115         }
116         else
117         {
118             this.type = type;
119             this.primitiveType = null;
120         }
121 
122         // Name, path, format and converter
123         this.name = name;
124         this.path = path;
125         if (format == null || format.length() == 0)
126         {
127             this.format = null;
128         }
129         else
130         {
131             this.format = format;
132         }
133         this.stripWsnl = stripWsnl;
134         this.converter = converter;
135 
136         // Reference type, property stuff, variable names and metadata
137         this.referenceType = referenceType;
138         this.getter = getter;
139         this.setter = setter;
140         this.variableNames = variableNames;
141         this.metadata = new HashMap<String, Object>();

150         StringBuilder builder = new StringBuilder().append(type.getParameterizedQualifiedSourceName()).append(" ")
