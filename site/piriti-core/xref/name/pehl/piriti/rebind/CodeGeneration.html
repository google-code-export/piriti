18      /**
19       * To ensure all necessary readers are registered, this little helper method
20       * genereates a new instance of the specified type (only if the type
21       * provides a noarg constructor).
22       * 
23       * <pre>
24       * new &lt;specified type&gt;();
25       * </pre>
26       * 
27       * @param writer
28       * @param type
29       */
30      public static void writeReaderInitialization(IndentedWriter writer, JClassType type)
31      {
32          boolean noargConstructor = false;
33          JConstructor[] constructors = type.getConstructors();
34          if (constructors != null)
35          {
36              for (JConstructor constructor : constructors)
37              {
38                  JParameter[] parameters = constructor.getParameters();
39                  if (parameters == null || parameters.length == 0)
40                  {
41                      noargConstructor = true;
42                      break;
43                  }
44              }
45          }
46          else
47          {
48              noargConstructor = true;
49          }
50          if (noargConstructor)
51          {
52              writer.write("new %s();", type.getParameterizedQualifiedSourceName());
53          }
54      }
