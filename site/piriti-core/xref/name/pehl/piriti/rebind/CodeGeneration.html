277      * To ensure all necessary reader and writer are registered, this little
278      * helper method genereates a new instance of the specified type (only if

288     public static void readerWriterInitialization(IndentedWriter writer, JClassType type)
289     {
290         boolean noargConstructor = false;
291         JConstructor[] constructors = type.getConstructors();
292         if (constructors != null)
293         {
294             for (JConstructor constructor : constructors)
295             {
296                 JParameter[] parameters = constructor.getParameters();
297                 if (parameters == null || parameters.length == 0)
298                 {
299                     noargConstructor = true;
300                     break;
301                 }
302             }
303         }
304         else
305         {
306             noargConstructor = true;
307         }
308         if (noargConstructor)

310             writer.write(
311                     "new %1$s(); // if there are any reader definitions in %1$s, this ensures they are registered",
312                     type.getParameterizedQualifiedSourceName());
313         }
