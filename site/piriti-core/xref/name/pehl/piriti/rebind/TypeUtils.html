44       * Returns true if the type is boolean or Boolean, false otherwise.
45       * 
46       * @param type
47       * @return true if the type is boolean or Boolean, false otherwise.
48       */
49      public static boolean isBoolean(JType type)
50      {
51          if (type != null)
52          {
53              JPrimitiveType primitive = type.isPrimitive();
54              return (primitive != null && primitive.equals(JPrimitiveType.BOOLEAN))
55                      || type.getQualifiedSourceName().equals(Boolean.class.getName());
56          }
57          return false;
58      }

306      * Returns the first type variable of the specified parameterized type or
307      * null if {@code type} is <code>null</code> or does not have any type
308      * variables.
309      * 
310      * @param type
311      * @return
312      */
313     public static JClassType getTypeVariable(JType type)
314     {
315         JClassType parameterType = null;
316         JParameterizedType parameterizedType = type.isParameterized();
317         if (parameterizedType != null)
318         {
319             JClassType[] typeArgs = parameterizedType.getTypeArgs();
320             if (typeArgs != null && typeArgs.length > 0)
321             {
322                 parameterType = typeArgs[0];
323             }
324         }
325         return parameterType;
326     }

329     /**
330      * Checks whether the given field is accessible in the given type or in the
331      * supertypes of the given type. If {@code readOnly} is <code>true</code>,
332      * the field is accesible if it isn't private. If {@code readOnly} is
333      * <code>false</code>, the field is accesible if it isn't private or final.
334      * 
335      * @param type
336      * @param name
337      * @param readOnly
338      * @return
339      */
340     public static boolean isFieldAccessible(JClassType type, String name, boolean readOnly)
341     {
342         boolean accessible = false;
343         if (type != null && name != null && name.length() != 0)
344         {
345             JField field = type.getField(name);
346             if (field != null)
347             {
348                 accessible = readOnly ? !field.isPrivate() : !(field.isPrivate() || field.isFinal());
349                 if (!accessible)
350                 {
351                     accessible = isFieldAccessible(type.getSuperclass(), name, readOnly);
352                 }
353             }
354             else
355             {
356                 accessible = isFieldAccessible(type.getSuperclass(), name, readOnly);
357             }
358         }
359         return accessible;
360     }

427     /**
428      * Checks whether the given method with the given parameters is accessible
429      * in the given type or in the supertypes of the given type. If the method
430      * you're looking for takes no arguments, use <code>null</code> for
431      * {@code parameter}.
432      * 
433      * @param type
434      * @param name
435      * @param parameter
436      *            The parameter of the method or <code>null</code> if the method
437      *            takes no parameter.
438      * @return
439      */
440     public static boolean isMethodAccessible(JClassType type, String name, JType parameter)
441     {
442         boolean accessible = false;
443         if (type != null && name != null && name.length() != 0)
444         {
445             try
446             {
447                 JType[] params = parameter == null ? new JType[0] : new JType[] {parameter};
448                 JMethod method = type.getMethod(name, params);
449                 if (method != null)
450                 {
451                     accessible = !method.isPrivate();
452                     if (!accessible)
453                     {
454                         // Autoboxiong: Try with primitive
455                         if (parameter != null && parameter.isPrimitive() != null)
456                         {
457                             params = new JType[] {parameter.isPrimitive()};
458                             method = type.getMethod(name, params);
459                             accessible = method != null && !method.isPrivate();
460                         }
461                         if (!accessible)
462                         {
463                             accessible = isMethodAccessible(type.getSuperclass(), name, parameter);
464                         }
465                     }
466                 }
467                 else
468                 {
469                     accessible = isMethodAccessible(type.getSuperclass(), name, parameter);
470                 }
471             }
472             catch (NotFoundException e)
473             {
474                 accessible = isMethodAccessible(type.getSuperclass(), name, parameter);
475             }
476         }
477         return accessible;
478     }
