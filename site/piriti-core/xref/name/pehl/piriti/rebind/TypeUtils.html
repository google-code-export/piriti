48      public static boolean isBoolean(JType type)
49      {
50          if (type != null)
51          {
52              JPrimitiveType primitive = type.isPrimitive();
53              return primitive != null && primitive.equals(JPrimitiveType.BOOLEAN)

305      * Returns the first type variable of the specified parameterized type or
306      * null if {@code type} is <code>null</code> or does not have any type
307      * variables.
308      * 
309      * @param type
310      * @return
311      */
312     public static JClassType getTypeVariable(JType type)
313     {
314         JClassType parameterType = null;
315         JParameterizedType parameterizedType = type.isParameterized();
316         if (parameterizedType != null)
317         {
318             JClassType[] typeArgs = parameterizedType.getTypeArgs();
319             if (typeArgs != null && typeArgs.length > 0)
320             {
321                 parameterType = typeArgs[0];
322             }
323         }
324         return parameterType;
325     }

328     /**
329      * Tries to find the given field with the given modifier in the given type
330      * or in the supertypes of the given type. If {@code modifiers} is
331      * <code>null</code>, the fields modifiers are not evaluated, otherwise the
332      * field must have at least one of the specified modifiers.
333      * 
334      * @param type
335      * @param property
336      * @param modifier
337      * @return the specified field or <code>null</code> if no such field was
338      *         found.
339      */
340     public static JField findField(JClassType type, String property, Modifier... modifiers)
341     {
342         JField field = null;
343         if (type != null && property != null && property.length() != 0)
344         {
345             field = type.getField(property);
346             if (field != null)
347             {
348                 boolean accessible = true;
349                 if (modifiers != null && modifiers.length != 0)
350                 {
351                     accessible = false;
352                     for (Modifier modifier : modifiers)
353                     {
354                         switch (modifier)
355                         {
356                             case PRIVATE:
357                                 accessible = field.isPrivate();
358                                 break;
359                             case DEFAULT:
360                                 accessible = field.isDefaultAccess();
361                                 break;
362                             case PROTECTED:
363                                 accessible = field.isProtected();
364                                 break;
365                             case PUBLIC:
366                                 accessible = field.isPublic();
367                                 break;
368                             default:
369                                 break;
370                         }
371                         if (accessible)
372                         {
373                             break;
374                         }
375                     }
376                 }
377                 if (!accessible)
378                 {
379                     field = findField(type.getSuperclass(), property, modifiers);
380                 }
381             }
382             else
383             {
384                 field = findField(type.getSuperclass(), property, modifiers);
385             }
386         }
387         return field;
388     }

391     public static JMethod findGetter(JClassType type, String property, JType returnType, Modifier... modifiers)
392     {
393         JMethod getter = null;
394         String[] methodNames = new String[] {"get" + property.substring(0, 1).toUpperCase() + property.substring(1),
395                 "is" + property.substring(0, 1).toUpperCase() + property.substring(1)};
396         for (String methodName : methodNames)
397         {
398             getter = findMethod(type, returnType, methodName, new JType[] {}, modifiers);
399             if (getter != null)
400             {
401                 break;
402             }
403         }
404         return getter;
405     }

415     /**
416      * Tries to find the given method with the given modifier in the given type
417      * or in the supertypes of the given type. If {@code modifiers} is
418      * <code>null</code>, the methods modifiers are not evaluated, otherwise the
419      * method must have at least one of the specified modifiers.
420      * 
421      * @param type
422      * @param returnType
423      * @param name
424      * @param parameter
425      * @param modifiers
426      * @return the specified method or <code>null</code> if no such method was
427      *         found.
428      */
429     private static JMethod findMethod(JClassType type, JType returnType, String name, JType[] parameter,
430             Modifier... modifiers)
431     {
432         JMethod method = null;
433         if (type != null && returnType != null && name != null && name.length() != 0 && parameter != null)
434         {
435             method = type.findMethod(name, parameter);
436             if (method != null && method.getReturnType() == returnType)
437             {
438                 boolean accessible = true;
439                 if (modifiers != null && modifiers.length != 0)
440                 {
441                     accessible = false;
442                     for (Modifier modifier : modifiers)
443                     {
444                         switch (modifier)
445                         {
446                             case PRIVATE:
447                                 accessible = method.isPrivate();
448                                 break;
449                             case DEFAULT:
450                                 accessible = method.isDefaultAccess();
451                                 break;
452                             case PROTECTED:
453                                 accessible = method.isProtected();
454                                 break;
455                             case PUBLIC:
456                                 accessible = method.isPublic();
457                                 break;
458                             default:
459                                 break;
460                         }
461                         if (accessible)
462                         {
463                             break;
464                         }
465                     }
466                 }
467                 if (!accessible)
468                 {
469                     method = findMethod(type.getSuperclass(), returnType, name, parameter, modifiers);
470                 }
471             }
472             else
473             {
474                 method = findMethod(type.getSuperclass(), returnType, name, parameter, modifiers);
475             }
476         }
477         return method;
478     }
