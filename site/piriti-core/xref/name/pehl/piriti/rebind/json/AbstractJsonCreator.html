99       * for {@link JsonField} as value.

103     private Map<String, FieldAnnotation<JsonField>> findFieldAnnotations()
104     {
105         Map<String, FieldAnnotation<JsonField>> fields = new HashMap<String, FieldAnnotation<JsonField>>();
106 
107         // Step 1: Add all JsonField annotations in the JsonFields annotation
108         // from the interfaceType
109         JsonFields interfaceTypeFields = interfaceType.getAnnotation(JsonFields.class);
110         if (interfaceTypeFields != null)
111         {
112             JsonField[] annotations = interfaceTypeFields.value();
113             for (JsonField annotation : annotations)
114             {
115                 JField field = modelType.getField(annotation.name());
116                 if (field != null)
117                 {
118                     fields.put(field.getName(), new FieldAnnotation<JsonField>(field, annotation,
119                             AssignmentPolicy.PROPERTY_FIRST));
120                 }
121                 // TODO Is it an error if field == null?
122             }
123         }
124 
125         // Step 2: Add all JsonField annotations of the modelType fields. If
126         // there's already an entry for the field from step 1, it will be
127         // overwritten!
128         JField[] modelTypeFields = findAnnotatedFields(modelType, JsonField.class);
129         for (JField field : modelTypeFields)
130         {
131             JsonField annotation = field.getAnnotation(JsonField.class);
132             fields.put(field.getName(), new FieldAnnotation<JsonField>(field, annotation, AssignmentPolicy.FIELD_ONLY));
133         }
134         return fields;
135     }
