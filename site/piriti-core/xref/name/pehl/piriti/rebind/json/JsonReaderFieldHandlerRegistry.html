11   * {@link FieldHandlerRegistry} used by the {@link JsonReaderCreator}.

13   * @author $LastChangedBy: harald.pehl $
14   * @version $LastChangedRevision: 136 $
15   */
16  public class JsonReaderFieldHandlerRegistry extends JsonFieldHandlerRegistry
17  {
18      /**
19       * Looks up a field handler based on the information provided in the field
20       * context. The lookup logic is implemented like this:
21       * <ol>
22       * <li>if the fields type is an enum return {@link #newEnumFieldHandler()}
23       * <li>If the fields type is an array return {@link #newArrayFieldHandler()}
24       * <li>If there's a format specified in the annotation and the type is int
25       * Integer, long, Long, float, Float, double or Double return
26       * {@link #newConverterFieldHandler()}
27       * <li>Try to lookup the field handler by the fields type classname (this
28       * will resolve all types registered in
29       * {@link #registerInitialFieldHandlers()}
30       * <li>If no field handler return {@link #newRegistryFieldHandler()}
31       * </ol>
32       * 
33       * @param fieldContext
34       * @return
35       * @see name.pehl.piriti.rebind.fieldhandler.FieldHandlerRegistry#findFieldHandler(name.pehl.piriti.rebind.fieldhandler.FieldContext)
36       */
37      public FieldHandler findFieldHandler(FieldContext fieldContext)
38      {
39          FieldHandler handler = null;
40          if (fieldContext.isEnum())
41          {
42              handler = newEnumFieldHandler();
43          }
44          else if (fieldContext.isArray())
45          {
46              handler = newArrayFieldHandler();
47          }
48          else
49          {
50              // If there's a format specified in the annotation and
51              // the type is int, long, float or double use the converter
52              // to convert the value, which is expected to be a string!
53              JType fieldType = fieldContext.getFieldType();
54              if (fieldContext.getFormat() != null
55                      && (TypeUtils.isInteger(fieldType) || TypeUtils.isLong(fieldType) || TypeUtils.isFloat(fieldType) || TypeUtils
56                              .isDouble(fieldType)))
57              {
58                  handler = newConverterFieldHandler();
59              }
60              else
61              {
62                  // Ask the registry for all other stuff (basic types,
63                  // collections, maps, ...)
64                  handler = registry.get(fieldContext.getFieldType().getQualifiedSourceName());
65                  if (handler == null)
66                  {
67                      // Delegate to the XmlRegistry to resolve other mapped
68                      // models
69                      handler = newRegistryFieldHandler();
70                  }
71              }
72          }
73          return handler;
74      }
