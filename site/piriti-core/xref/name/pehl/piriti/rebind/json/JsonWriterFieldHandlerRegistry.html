15      /**
16       * Looks up a field handler based on the information provided in the field
17       * context. The lookup logic is implemented like this:
18       * <ol>
19       * <li>if the fields type is an enum return {@link #newEnumFieldHandler()}
20       * <li>If the fields type is an array return {@link #newArrayFieldHandler()}
21       * <li>Try to lookup the field handler by the fields type classname (this
22       * will resolve all types registered in
23       * {@link #registerInitialFieldHandlers()}
24       * <li>If no field handler return {@link #newRegistryFieldHandler()}
25       * </ol>
26       * 
27       * @param fieldContext
28       * @return
29       * @see name.pehl.piriti.rebind.fieldhandler.FieldHandlerRegistry#findFieldHandler(name.pehl.piriti.rebind.fieldhandler.FieldContext)
30       */
31      public FieldHandler findFieldHandler(FieldContext fieldContext)
32      {
33          FieldHandler handler = null;
34          if (fieldContext.isEnum())
35          {
36              handler = newEnumFieldHandler();
37          }
38          else if (fieldContext.isArray())
39          {
40              handler = newArrayFieldHandler();
41          }
42          else
43          {
44              // Ask the registry for all other stuff (basic types,
45              // collections, maps, ...)
46              handler = registry.get(fieldContext.getFieldType().getQualifiedSourceName());
47              if (handler == null)
48              {
49                  // Delegate to the JsonRegistry to resolve other mapped
50                  // models
51                  handler = newRegistryFieldHandler();
52              }
53          }
54          return handler;
55      }
