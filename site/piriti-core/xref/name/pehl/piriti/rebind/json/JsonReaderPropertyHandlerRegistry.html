11   * {@link PropertyHandlerRegistry} used by the {@link JsonReaderCreator}.
12   * 
13   * @author $LastChangedBy: harald.pehl $
14   * @version $LastChangedRevision: 136 $
15   */
16  public class JsonReaderPropertyHandlerRegistry extends JsonPropertyHandlerRegistry
17  {
18      /**
19       * Looks up a property handler based on the information provided in the
20       * property context. The lookup logic is implemented like this:
21       * <ol>
22       * <li>if the properties type is an enum return
23       * {@link #newEnumFieldHandler()}
24       * <li>If the properties type is an array return
25       * {@link #newArrayFieldHandler()}
26       * <li>If there's a format specified in the annotation and the type is int
27       * Integer, long, Long, float, Float, double or Double return
28       * {@link #newConverterFieldHandler()}
29       * <li>Try to lookup the property handler by the properties type classname
30       * (this will resolve all types registered in
31       * {@link #registerInitialPropertyHandlers()}
32       * <li>Return {@link #newRegistryFieldHandler()}
33       * </ol>
34       * 
35       * @param propertyContext
36       * @return
37       * @see name.pehl.piriti.rebind.propertyhandler.PropertyHandlerRegistry#findPropertyHandler(name.pehl.piriti.rebind.propertyhandler.PropertyContext)
38       */
39      @Override
40      public PropertyHandler findPropertyHandler(PropertyContext propertyContext)
41      {
42          PropertyHandler handler = null;
43          if (propertyContext.isEnum())
44          {
45              handler = newEnumFieldHandler();
46          }
47          else if (propertyContext.isArray())
48          {
49              handler = newArrayFieldHandler();
50          }
51          else
52          {
53              // If there's a format or converter specified in the annotation and
54              // the type is int, long, float or double use the converter to
55              // convert the value, which is expected to be a string!
56              JType fieldType = propertyContext.getType();
57              if ((propertyContext.getFormat() != null || propertyContext.isCustomConverter())
58                      && (TypeUtils.isInteger(fieldType) || TypeUtils.isLong(fieldType) || TypeUtils.isFloat(fieldType) || TypeUtils
59                              .isDouble(fieldType)))
60              {
61                  handler = newConverterFieldHandler();
62              }
63              else
64              {
65                  // Ask the registry for all other stuff (basic types,
66                  // collections, maps, ...)
67                  handler = registry.get(propertyContext.getType().getQualifiedSourceName());
68                  if (handler == null)
69                  {
70                      // Delegate to the JsonRegistry to resolve other mapped
71                      // models
72                      handler = newRegistryFieldHandler();
73                  }
74              }
75          }
76          return handler;
77      }
