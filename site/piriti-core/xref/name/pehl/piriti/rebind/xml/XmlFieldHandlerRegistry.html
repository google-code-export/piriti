116     /**
117      * Looks up a field handler based on the information provided in the field
118      * context. The lookup logic is implemented like this:
119      * <ol>
120      * <li>If the assignment type is {@link AssignmentType#ID} return
121      * {@link IdFieldHandler}
122      * <li>if the assignment type is {@link AssignmentType#IDREF} return
123      * {@link IdRefFieldHandler}
124      * <li>if the fields type is a primitive return the
125      * {@link ConverterFieldHandler}
126      * <li>if the fields type is an enum return {@link EnumFieldHandler}
127      * <li>If the fields type is an array return {@link ArrayFieldHandler}
128      * <li>Try to lookup the field handler by the fields type classname (this
129      * will resolve all types registered in
130      * {@link #registerInitialFieldHandlers()}
131      * <li>If no field handler return {@link XmlRegistryFieldHandler}.
132      * </ol>
133      * 
134      * @param fieldContext
135      * @return
136      * @see name.pehl.piriti.rebind.fieldhandler.FieldHandlerRegistry#findFieldHandler(name.pehl.piriti.rebind.fieldhandler.FieldContext)
137      */
138     public FieldHandler findFieldHandler(FieldContext fieldContext)
139     {
140         FieldHandler handler = null;
141 
142         if (fieldContext.getAssignmentType() == AssignmentType.ID)
143         {
144             handler = newIdFieldHandler();
145         }
146         else if (fieldContext.getAssignmentType() == AssignmentType.IDREF)
147         {
148             handler = newIdRefFieldHandler();
149         }
150         else if (fieldContext.isPrimitive())
151         {
152             handler = newConverterFieldHandler();
153         }
154         else if (fieldContext.isEnum())
155         {
156             handler = newEnumFieldHandler();
157         }
158         else if (fieldContext.isArray())
159         {
160             handler = newArrayFieldHandler();
161         }
162         else
163         {
164             // Ask the registry for all other stuff (basic types,
165             // collections, maps, ...)
166             handler = registry.get(fieldContext.getFieldType().getQualifiedSourceName());
167             if (handler == null)
168             {
169                 // Delegate to the XmlRegistry to resolve other mapped models
170                 handler = newRegistryFieldHandler();
171             }
172         }
173         return handler;
174     }
