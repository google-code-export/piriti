26      /**
27       * Returns <code>true</code> if the field type is a class or interface and
28       * if there's a public static field of type {@link #getReaderClassname()} in
29       * the field type or if the field type is an array or collection of such a
30       * type, <code>false</code> otherwise.
31       * 
32       * @param writer
33       * @param fieldContext
34       * @return
35       * @see name.pehl.piriti.rebind.fieldhandler.AbstractFieldHandler#isValid(name.pehl.piriti.rebind.fieldhandler.FieldContext)
36       */
37      @Override
38      public boolean isValid(IndentedWriter writer, FieldContext fieldContext) throws UnableToCompleteException
39      {
40          JType type = null;
41          if (fieldContext.isArray())
42          {
43              type = fieldContext.getArrayType().getComponentType();
44          }
45          else if (TypeUtils.isCollection(fieldContext.getFieldType()))
46          {
47              type = TypeUtils.getTypeVariable(fieldContext.getFieldType());
48          }
49          else
50          {
51              type = fieldContext.getFieldType();
52          }
53  
54          JClassType classOrInterface = type.isClassOrInterface();
55          if (classOrInterface == null)
56          {
57              skipField(writer, fieldContext, "Type is no class or interface");
58              return false;
59          }
60          JField xmlReaderField = findReaderMember(classOrInterface);
61          if (xmlReaderField == null)
62          {
63              skipField(writer, fieldContext, String.format("No public static field of type %1$s<%2$s> found in %2$s",
64                      getReaderClassname(), type.getQualifiedSourceName()));
65              return false;
66          }
67          fieldContext.addMetadata(NESTED_TYPE, classOrInterface);
68          fieldContext.addMetadata(FQ_XML_READER, classOrInterface.getQualifiedSourceName() + "."
69                  + xmlReaderField.getName());
70          return true;
71      }

74      @Override
75      public void writeConverterCode(IndentedWriter writer, FieldContext fieldContext) throws UnableToCompleteException
76      {
77          JClassType nestedType = fieldContext.getMetadata(NESTED_TYPE);
78          String fqXmlReader = fieldContext.getMetadata(FQ_XML_READER);
79  
80          String references = fieldContext.newVariableName("References");
81          writer.write("String[] %s = %s.selectValues(\"%s\", %s);", references, fieldContext.getInputVariable(),
82                  fieldContext.getPath(), fieldContext.isStripWsnl());
83          writer.write("if (%s.length == 1) {", references);
84          writer.indent();
85          // If there's only one value it is expected that this value contains
86          // the references seperated with space
87          writer.write("%1$s = %1$s[0].split(\" \");", references);
88          writer.outdent();
89          writer.write("}");
90          writer.write("if (%s.length != 0) {", references);
91          writer.indent();
92          if (fieldContext.isArray() || TypeUtils.isCollection(fieldContext.getFieldType()))
93          {
94              String collectionVariable = null;
95              if (fieldContext.isArray())
96              {
97                  collectionVariable = fieldContext.newVariableName("AsList");
98                  writer.write("List<%1$s> %2$s = new ArrayList<%1$s>();", nestedType
99                          .getParameterizedQualifiedSourceName(), collectionVariable);
100             }
101             else
102             {
103                 collectionVariable = fieldContext.getValueVariable();
104                 String collectionImplementation = AbstractCollectionFieldHandler.interfaceToImplementation
105                         .get(fieldContext.getFieldType().getErasedType().getQualifiedSourceName());
106                 if (collectionImplementation == null)
107                 {
108                     // the field type is already an implementation
109                     collectionImplementation = fieldContext.getFieldType().getParameterizedQualifiedSourceName();
110                 }
111                 writer.write("%s = new %s<%s>();", fieldContext.getValueVariable(), collectionImplementation,
112                         nestedType.getQualifiedSourceName());
113             }
114             writer.write("for (String reference : %s) {", references);
115             writer.indent();
116             writer.write("%s referenceInstance = %s.idRef(reference);", nestedType
117                     .getParameterizedQualifiedSourceName(), fqXmlReader);
118             writer.write("if (referenceInstance != null) {");
119             writer.indent();
120             writer.write("%s.add(referenceInstance);", collectionVariable);
121             writer.outdent();
122             writer.write("}");
123             writer.outdent();
124             writer.write("}");
125             if (fieldContext.isArray())
126             {
127                 writer.write("%s = new %s[%s.size()];", fieldContext.getValueVariable(), nestedType
128                         .getQualifiedSourceName(), collectionVariable);
129                 writer.write("int index = 0;");
130                 writer.write("for(%s currentValue : %s) {", nestedType.getQualifiedSourceName(), collectionVariable);
131                 writer.indent();
132                 writer.write("%s[index] = currentValue;", fieldContext.getValueVariable());
133                 writer.write("index++;");
134                 writer.outdent();
135                 writer.write("}");
136             }
137         }
138         else
139         {
140             writer.write("%s = %s.idRef(%s[0]);", fieldContext.getValueVariable(), fqXmlReader, references);
141         }
142         writer.outdent();
143         writer.write("}");
144     }
