113     protected Map<String, FieldAnnotation<XmlField>> findFieldAnnotations()
114     {
115         Map<String, FieldAnnotation<XmlField>> fields = new HashMap<String, FieldAnnotation<XmlField>>();
116 
117         // Step 1: Add all XmlField annotations in the XmlFields annotation
118         // from the interfaceType
119         XmlFields xmlFields = interfaceType.getAnnotation(XmlFields.class);
120         if (xmlFields != null)
121         {
122             XmlField[] annotations = xmlFields.value();
123             for (XmlField annotation : annotations)
124             {
125                 JField field = modelType.getField(annotation.name());
126                 if (field != null)
127                 {
128                     fields.put(field.getName(), new FieldAnnotation<XmlField>(field, annotation,
129                             AssignmentPolicy.PROPERTY_FIRST));
130                 }
131                 // TODO Is it an error if field == null?
132             }
133         }
134 
135         // Step 2: Add all XmlField annotations of the modelType fields. If
136         // there's already an entry for the field from step 1, it will be
137         // overwritten!
138         JField[] modelTypeFields = findAnnotatedFields(modelType, XmlField.class);
139         for (JField field : modelTypeFields)
140         {
141             XmlField annotation = field.getAnnotation(XmlField.class);
142             fields.put(field.getName(), new FieldAnnotation<XmlField>(field, annotation, AssignmentPolicy.FIELD_ONLY));
143         }
144         return fields;

148     protected String calculateXpath(JField field, String defaultValue)
149     {
150         String xpath = defaultValue;
151         if (xpath == null || xpath.length() == 0)
