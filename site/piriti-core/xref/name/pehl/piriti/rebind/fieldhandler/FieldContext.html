45       * Construct a new instance of this class
46       * 
47       * @param typeOracle
48       *            The type oracle from the GWT generator API
49       * @param handlerRegistry
50       *            The handler registry for {@link FieldHandler} lookup
51       * @param modelType

59       * @param format

61       * @param assignmentType

63       * @param inputVariable
64       *            The name of the input variable
65       * @param valueVariable
66       *            The name of the value variable
67       * @throws UnableToCompleteException
68       */
69      public FieldContext(TypeOracle typeOracle, FieldHandlerRegistry handlerRegistry, JClassType modelType,
70              JType fieldType, String fieldName, String path, String format, boolean stripWsnl,
71              AssignmentType assignmentType, String inputVariable, String valueVariable) throws UnableToCompleteException
72      {
73          // Types
74          this.typeOracle = typeOracle;
75          this.handlerRegistry = handlerRegistry;
76          this.modelType = modelType;
77          JPrimitiveType primitiveType = fieldType.isPrimitive();
78          if (primitiveType != null) // isPrimitive() is not yet available!
79          {
80              try
81              {
82                  // Use the boxed type for primitives
83                  this.fieldType = typeOracle.getType(primitiveType.getQualifiedBoxedSourceName());
84              }
85              catch (NotFoundException e)
86              {
87                  throw new UnableToCompleteException();
88              }
89          }
90          else
91          {
92              this.fieldType = fieldType;
93          }
94  
95          // Field properties
96          this.fieldName = fieldName;
97          this.path = path;
98          if (format == null || format.length() == 0)
99          {
100             this.format = null;
101         }
102         else
103         {
104             this.format = format;
105         }
106         this.stripWsnl = stripWsnl;
107         this.assignmentType = assignmentType;
108 
109         // Variable names
110         this.inputVariable = inputVariable;
111         this.valueVariable = valueVariable;
112         this.metadata = new HashMap<String, Object>();
113     }

116     @Override
117     public String toString()
118     {
119         StringBuilder builder = new StringBuilder().append(fieldType.getParameterizedQualifiedSourceName()).append(" ")
120                 .append(fieldName).append(", path=\"").append(path).append("\"");
121         if (format != null)
122         {
123             builder.append(", format=\"").append(format).append("\"");
124         }
125         return builder.toString();
126     }
