38      /**
39       * Returns <code>false</code> if the field type is no collection, if the
40       * collection has no type arguments or if the type argument of the
41       * collection is an array, collection or map, <code>true</code> otherwise.
42       * 
43       * @param writer
44       * @param fieldContext
45       * @return
46       * @see name.pehl.piriti.rebind.fieldhandler.AbstractFieldHandler#isValid(name.pehl.piriti.rebind.fieldhandler.FieldContext)
47       */
48      @Override
49      public boolean isValid(IndentedWriter writer, FieldContext fieldContext) throws UnableToCompleteException
50      {
51          if (!fieldContext.isCollection())
52          {
53              CodeGeneration.skipField(writer, fieldContext, "Type is no collection");
54              return false;
55          }
56          JClassType parameterType = getTypeVariable(fieldContext);
57          if (parameterType != null)
58          {
59              if (parameterType.isArray() != null || TypeUtils.isCollection(parameterType)
60                      || TypeUtils.isMap(parameterType))
61              {
62                  CodeGeneration.skipField(writer, fieldContext, "Nested arrays / collections / maps are not supported");
63                  return false;
64              }
65          }
66          else
67          {
68              // collections and maps without type arguments are not
69              // supported!
70              CodeGeneration.skipField(writer, fieldContext, "Collection has no / invalid type argument");
71              return false;
72          }
73          // Initialize the parameter type to make sure the relevant reader
74          // is in the registry (ugly - but it works)
75          CodeGeneration.readerInitialization(writer, parameterType);
76          return true;
77      }
