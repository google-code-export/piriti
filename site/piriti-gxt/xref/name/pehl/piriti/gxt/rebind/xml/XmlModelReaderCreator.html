54      @Override
55      protected void handleProperties(IndentedWriter writer) throws UnableToCompleteException
56      {
57          int counter = 0;
58          Xml[] properties = findGxtPropertyAnnotations();
59          for (int i = 0; i < properties.length; i++)
60          {
61              Xml xmlField = properties[i];
62              writer.newline();
63              JClassType fieldType = getFieldType(xmlField);
64              String xpath = calculateXpath(fieldType, xmlField);
65              VariableNames variableNames = new VariableNames("element", "value" + counter, "xmlBuilder");
66              PropertyContext fieldContext = new PropertyContext(context.getTypeOracle(), handlerRegistry, interfaceType,
67                      modelType, fieldType, xmlField.property(), xpath, xmlField.format(), xmlField.stripWsnl(),
68                      xmlField.converter(), null, NoopPropertyGetter.class, NoopPropertySetter.class, variableNames);
69              fieldContext.addMetadata(TYPE_VARIABLE, xmlField.typeVariable());
70              PropertyHandler fieldHandler = handlerRegistry.findPropertyHandler(fieldContext);
71              if (fieldHandler != null && fieldHandler.isValid(writer, fieldContext))
72              {
73                  writer.newline();
74                  handleProperty(writer, fieldHandler, fieldContext, (i < properties.length - 1));
75                  counter++;
76              }
77          }
78      }

81      private Xml[] findGxtPropertyAnnotations()
82      {
83          Map<String, Xml> properties = new HashMap<String, Xml>();
84  
85          // Step 1: Add all XmlField annotations from the interfaceType
86          XmlMappings interfaceTypeFields = interfaceType.getAnnotation(XmlMappings.class);
87          if (interfaceTypeFields != null)
88          {
89              Xml[] annotations = interfaceTypeFields.value();
90              for (Xml annotation : annotations)
91              {
92                  properties.put(annotation.property(), annotation);
93              }
94          }
95  
96          // Step 2: Add all XmlField annotations from the modelType. If
97          // there's already an entry from step 1, it will be overwritten!
98          collectModelTypeFields(modelType, properties);
99  
100         return properties.values().toArray(new Xml[] {});
101     }

104     private void collectModelTypeFields(JClassType type, Map<String, Xml> fields)
105     {
106         // Superclass first please!
107         if (type == null)
108         {
109             return;
110         }
111         collectModelTypeFields(type.getSuperclass(), fields);
112 
113         XmlMappings modelTypeFields = type.getAnnotation(XmlMappings.class);
114         if (modelTypeFields != null)
115         {
116             Xml[] modelTypeFieldsValue = modelTypeFields.value();
117             if (modelTypeFieldsValue != null)
118             {
119                 for (Xml annotation : modelTypeFieldsValue)
120                 {
121                     fields.put(annotation.property(), annotation);
122                 }
123             }
124         }
125     }

128     private JClassType getFieldType(Xml xmlField) throws UnableToCompleteException
129     {
130         JClassType fieldType = null;
131         if (xmlField.array())
132         {
133             JClassType componentType = context.getTypeOracle().findType(xmlField.type().getName());
134             fieldType = context.getTypeOracle().getArrayType(componentType);
135         }
136         else
137         {
138             fieldType = context.getTypeOracle().findType(xmlField.type().getName());
139         }
140         if (fieldType == null)
141         {
142             die("Cannot find type {0}", xmlField.type().getName());
143         }
144         return fieldType;
145     }

148     private String calculateXpath(JClassType fieldType, Xml xmlField)
149     {
150         String xpath = xmlField.path();
151         if (xpath == null || xpath.length() == 0)
152         {
153             xpath = xmlField.property();
154             if (fieldType.isPrimitive() != null || TypeUtils.isBasicType(fieldType) || fieldType.isEnum() != null)
155             {
156                 xpath += "/text()";
157             }
158         }
159         return xpath;
