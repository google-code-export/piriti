58      @Override
59      protected void handleProperties(IndentedWriter writer) throws UnableToCompleteException
60      {
61          int counter = 0;
62          Json[] properties = findGxtPropertyAnnotations();
63          for (int i = 0; i < properties.length; i++)
64          {
65              Json jsonField = properties[i];
66              writer.newline();
67              JClassType fieldType = getFieldType(jsonField);
68              String jsonPath = calculateJsonPath(jsonField);
69              VariableNames variableNames = new VariableNames("jsonObject", "value" + counter, "jsonBuilder");
70              PropertyContext fieldContext = new PropertyContext(context.getTypeOracle(), handlerRegistry, interfaceType,
71                      modelType, fieldType, jsonField.property(), jsonPath, jsonField.format(), false,
72                      jsonField.converter(), null, NoopPropertyGetter.class, NoopPropertySetter.class, variableNames);
73              fieldContext.addMetadata(TYPE_VARIABLE, jsonField.typeVariable());
74              PropertyHandler fieldHandler = handlerRegistry.findPropertyHandler(fieldContext);
75              if (fieldHandler != null && fieldHandler.isValid(writer, fieldContext))
76              {
77                  writer.newline();
78                  handleProperty(writer, fieldHandler, fieldContext, (i < properties.length - 1));
79                  counter++;
80              }
81          }
82      }

85      private Json[] findGxtPropertyAnnotations()
86      {
87          Map<String, Json> properties = new HashMap<String, Json>();
88  
89          // Step 1: Add all JsonField annotations from the interfaceType
90          JsonMappings interfaceTypeFields = interfaceType.getAnnotation(JsonMappings.class);
91          if (interfaceTypeFields != null)
92          {
93              Json[] annotations = interfaceTypeFields.value();
94              for (Json annotation : annotations)
95              {
96                  properties.put(annotation.property(), annotation);
97              }
98          }
99  
100         // Step 2: Add all JsonField annotations from the modelType. If
101         // there's already an entry from step 1, it will be overwritten!
102         collectModelTypeFields(modelType, properties);
103 
104         return properties.values().toArray(new Json[] {});
105     }

108     private void collectModelTypeFields(JClassType type, Map<String, Json> fields)
109     {
110         // Superclass first please!
111         if (type == null)
112         {
113             return;
114         }
115         collectModelTypeFields(type.getSuperclass(), fields);
116 
117         JsonMappings modelTypeFields = type.getAnnotation(JsonMappings.class);
118         if (modelTypeFields != null)
119         {
120             Json[] modelTypeFieldsValue = modelTypeFields.value();
121             if (modelTypeFieldsValue != null)
122             {
123                 for (Json annotation : modelTypeFieldsValue)
124                 {
125                     fields.put(annotation.property(), annotation);
126                 }
127             }
128         }
129     }

132     private JClassType getFieldType(Json jsonField) throws UnableToCompleteException
133     {
134         JClassType fieldType = null;
135         if (jsonField.array())
136         {
137             JClassType componentType = context.getTypeOracle().findType(jsonField.type().getName());
138             fieldType = context.getTypeOracle().getArrayType(componentType);
139         }
140         else
141         {
142             fieldType = context.getTypeOracle().findType(jsonField.type().getName());
143         }
144         if (fieldType == null)
145         {
146             die("Cannot find type {0}", jsonField.type().getName());
147         }
148         return fieldType;
149     }

152     private String calculateJsonPath(Json jsonField)
