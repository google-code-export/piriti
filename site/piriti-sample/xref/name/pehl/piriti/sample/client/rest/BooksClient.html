31      public <T> void readFromJson(final JsonReader<T> jsonReader, final String sourceCode)
32      {
33          ClientResource clientResource = new ClientResource(URL);
34          clientResource.setOnResponse(new Uniform()
35          {
36              @Override
37              public void handle(Request request, Response response)
38              {
39                  BooksEvent booksEvent = null;
40                  StopWatch stopWatch = new StopWatch();
41                  stopWatch.start();
42                  PiritiJsonRepresentation<T> representation = new PiritiJsonRepresentation<T>(jsonReader, response
43                          .getEntity());
44                  try
45                  {
46                      List<T> books = representation.getModels();
47                      TimeInterval timeInterval = stopWatch.stop();
48                      booksEvent = new BooksEvent(books, timeInterval, sourceCode);
49                  }
50                  catch (IOException e)
51                  {
52                      booksEvent = new BooksEvent(null, null, null);
53                  }
54                  EventBus.get().fireEvent(booksEvent);
55              }
56          });
57          clientResource.get(MediaType.APPLICATION_JSON);
58      }
