146     private static void createElementsAndAppend(Document document, Element parent, String elementName, String... values)
147     {
148         if (values != null && values.length != 0)
149         {
150             for (String value : values)
151             {
152                 createElementAndAppend(document, parent, elementName, value);
153             }
154         }
155     }

157 
158     private static Element createElementAndAppend(Document document, Element parent, String elementName, String value)
159     {
160         Element element = document.createElement(elementName);
161         Text text = document.createTextNode(value);
162         element.appendChild(text);
163         parent.appendChild(element);
164         return element;
165     }
