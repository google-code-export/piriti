65      /**
66       * Registers the specified converter. If the clazz is {@code null} no
67       * converter is registered. If a converter for the specified class is
68       * already registered it is overwritten with the ne converter.
69       */
70      @Override
71      public <T> void register(Class<T> clazz, Converter<T> converter)
72      {
73          if (clazz != null)
74          {
75              registry.put(clazz, converter);
76          }
77      }
